    <script>
        // ─── INIT ───────────────────────────────────────────
        document.getElementById('todayDate').textContent = new Date().toLocaleDateString('tr-TR',{weekday:'long',year:'numeric',month:'long',day:'numeric'});

        // ─── SIDEBAR TOGGLE ──────────────────────────────────
        const sidebar = document.getElementById('sidebar');
        const backdrop = document.getElementById('backdrop');
        let isMobile = () => window.innerWidth < 992;

        function toggleSidebar() {
          if (isMobile()) {
            sidebar.classList.toggle('mobile-open');
            backdrop.classList.toggle('show');
          } else {
            sidebar.classList.toggle('collapsed');
          }
        }
        function closeSidebar() {
          sidebar.classList.remove('mobile-open');
          backdrop.classList.remove('show');
        }
        window.addEventListener('resize', () => {
          if (!isMobile()) {
            sidebar.classList.remove('mobile-open');
            backdrop.classList.remove('show');
          }
        });

        // ─── NAVIGATION ──────────────────────────────────────
        const pageLabels = {
          dashboard:'Gösterge Paneli', doktorlar:'Doktorlar', branslar:'Branş Yönetimi',
          randevular:'Randevular', receteler:'Reçeteler', yorumlar:'Yorumlar',
          mesajlar:'Mesajlar', kullanicilar:'Kullanıcılar', ayarlar:'Ayarlar'
        };

        function navigate(page, el) {
          document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
          const target = document.getElementById('page-' + page);
          if (target) target.classList.add('active');

          document.querySelectorAll('.nav-item').forEach(i => i.classList.remove('active'));
          if (el) el.classList.add('active');
          else {
            document.querySelectorAll('.nav-item').forEach(i => {
              if (i.getAttribute('onclick') && i.getAttribute('onclick').includes("'"+page+"'")) i.classList.add('active');
            });
          }

          document.getElementById('breadcrumb').textContent = pageLabels[page] || page;
          if (isMobile()) closeSidebar();
        }

        // ─── MODALS ──────────────────────────────────────────
        function openModal(type, branchName) {
          if (type === 'editBranch') {
            document.getElementById('modalTitle').textContent = 'Branşı Düzenle';
            document.getElementById('modalBranchName').value = branchName || '';
            const descs = {
              'Kardiyoloji': 'Kalp ve damar hastalıklarının kapsamlı teşhis, tedavi ve takip hizmetleri sunulmaktadır.',
              'Dermatoloji': 'Cilt, saç ve tırnak hastalıklarının tanı ve tedavisi ile estetik dermatoloji uygulamaları.',
              'Ortopedi ve Travmatoloji': 'Kemik, eklem ve kas-iskelet sistemi hastalıkları ile spor yaralanmalarının tedavisi.',
              'Nöroloji': 'Beyin, omurilik ve sinir sistemi hastalıklarının ileri teşhis ve tedavi hizmetleri.',
              'Pediatri': '0–18 yaş arası çocukların büyüme, gelişme ve sağlık takipleri ile hastalık tedavileri.',
              'Radyoloji': 'MR, BT, ultrason ve dijital röntgen ile ileri düzey tanı görüntüleme hizmetleri.'
            };
            document.getElementById('modalBranchDesc').value = descs[branchName] || '';
            showOverlay('editModal');
          } else if (type === 'addBranch') {
            document.getElementById('addBranchName').value = '';
            document.getElementById('addBranchDesc').value = '';
            showOverlay('addModal');
          }
        }

        function showOverlay(id) {
          document.getElementById(id).classList.add('show');
          document.body.style.overflow = 'hidden';
        }
        function closeModal(id) {
          document.getElementById(id).classList.remove('show');
          document.body.style.overflow = '';
        }
        function handleOverlayClick(e, id) {
          if (e.target === document.getElementById(id)) closeModal(id);
        }
        function saveModal() {
          const name = document.getElementById('modalBranchName').value.trim();
          if (!name) { showToast('Branş adı boş bırakılamaz!','danger','bi-exclamation-circle'); return; }
          closeModal('editModal');
          showToast(name + ' branşı başarıyla güncellendi!','success','bi-check-circle');
        }
        function saveNewBranch() {
          const name = document.getElementById('addBranchName').value.trim();
          if (!name) { showToast('Branş adı boş bırakılamaz!','danger','bi-exclamation-circle'); return; }
          closeModal('addModal');
          showToast(name + ' branşı başarıyla eklendi!','success','bi-plus-circle');
        }

        let deletingBranch = '';
        function confirmDelete(name) {
          deletingBranch = name;
          document.getElementById('deleteBranchName').textContent = name;
          showOverlay('deleteModal');
        }
        function executDelete() {
          closeModal('deleteModal');
          showToast(deletingBranch + ' branşı silindi.','danger','bi-trash3');
        }

        // ─── TOAST ───────────────────────────────────────────
        function showToast(msg, type = 'success', icon = 'bi-check-circle') {
          const c = document.getElementById('toastContainer');
          const t = document.createElement('div');
          t.className = `toast-item ${type}`;
          t.innerHTML = `<i class="bi ${icon} ${type}"></i><span style="flex:1">${msg}</span><i class="bi bi-x" style="cursor:pointer;color:var(--text-muted);font-size:14px;" onclick="this.parentElement.remove()"></i>`;
          c.appendChild(t);
          setTimeout(() => { t.style.opacity='0'; t.style.transform='translateX(40px)'; t.style.transition='.3s ease'; setTimeout(()=>t.remove(),300); }, 3500);
        }

        // ─── BRANCH FILTER ───────────────────────────────────
        function filterBranch(cat, btn) {
          document.querySelectorAll('[onclick^="filterBranch"]').forEach(b => {
            b.className = 'btn-outline-custom';
            b.style.cssText = 'padding:7px 16px;font-size:12px;';
          });
          btn.className = 'btn-primary-custom';
          btn.style.cssText = 'padding:8px 16px;font-size:12px;';

          document.querySelectorAll('.branch-item').forEach(item => {
            if (cat === 'all' || item.dataset.cat === cat) {
              item.style.display = '';
              item.style.animation = 'fadeUp .25s ease';
            } else {
              item.style.display = 'none';
            }
          });
          const visible = document.querySelectorAll('.branch-item:not([style*="display: none"])').length;
          showToast(visible + ' branş görüntüleniyor.', 'info', 'bi-filter');
        }

        // ─── SEARCH ──────────────────────────────────────────
        let searchTimeout;
        function handleSearch(val) {
          clearTimeout(searchTimeout);
          if (!val.trim()) return;
          searchTimeout = setTimeout(() => {
            showToast('"' + val + '" için arama yapıldı.', 'info', 'bi-search');
          }, 600);
        }

        // ─── KEYBOARD ────────────────────────────────────────
        document.addEventListener('keydown', e => {
          if (e.key === 'Escape') {
            ['editModal','addModal','deleteModal'].forEach(id => {
              if (document.getElementById(id).classList.contains('show')) closeModal(id);
            });
          }
        });
