<script>
/* ─── Utility ───────────────────────────────────────── */
const $ = id => document.getElementById(id);
const setError = (fcId, show) => {
  const el = $(fcId);
  el.classList.toggle('has-error', show);
  el.classList.toggle('has-success', !show);
};
const clearState = fcId => {
  const el = $(fcId);
  el.classList.remove('has-error', 'has-success');
};

/* ─── Max birth date: today ─────────────────────────── */
$('dogum').max = new Date().toISOString().split('T')[0];

/* ─── Profile photo preview ─────────────────────────── */
$('file-input').addEventListener('change', e => {
  const file = e.target.files[0];
  if (!file) return;
  const reader = new FileReader();
  reader.onload = ev => { $('preview-img').src = ev.target.result; };
  reader.readAsDataURL(file);
});

/* ─── Phone masking ─────────────────────────────────── */
$('telefon').addEventListener('input', e => {
  let v = e.target.value.replace(/\D/g,'');
  if (v.startsWith('0')) {
    let p = v.slice(1);
    let out = '0';
    if (p.length > 0) out += p.slice(0,3);
    if (p.length > 3) out += ' ' + p.slice(3,6);
    if (p.length > 6) out += ' ' + p.slice(6,8);
    if (p.length > 8) out += ' ' + p.slice(8,10);
    e.target.value = out;
  }
});

/* ─── Password toggle ───────────────────────────────── */
function togglePw(inputId, iconId) {
  const inp = $(inputId), icon = $(iconId);
  if (inp.type === 'password') { inp.type = 'text'; icon.className = 'bi bi-eye-slash field-icon-right'; }
  else { inp.type = 'password'; icon.className = 'bi bi-eye field-icon-right'; }
}

/* ─── Password strength ─────────────────────────────── */
$('sifre').addEventListener('input', function() {
  const val = this.value;
  const pw = $('pw-strength');
  if (!val) { pw.style.display = 'none'; return; }
  pw.style.display = 'block';

  let score = 0;
  if (val.length >= 6) score++;
  if (val.length >= 10) score++;
  if (/[A-Z]/.test(val)) score++;
  if (/[0-9]/.test(val)) score++;
  if (/[^A-Za-z0-9]/.test(val)) score++;

  const fill = $('pw-fill');
  const label = $('pw-label');
  const configs = [
    { w:'20%', bg:'#e74c6e', txt:'Çok Zayıf' },
    { w:'40%', bg:'#f39c12', txt:'Zayıf' },
    { w:'60%', bg:'#f1c40f', txt:'Orta' },
    { w:'80%', bg:'#3ab3b3', txt:'Güçlü' },
    { w:'100%', bg:'#27ae8a', txt:'Çok Güçlü' },
  ];
  const c = configs[score - 1] || configs[0];
  fill.style.width = c.w;
  fill.style.background = c.bg;
  label.textContent = c.txt;
  label.style.color = c.bg;

  /* live step bar */
  updateStepBar();
});

/* ─── Step bar live update ──────────────────────────── */
function updateStepBar() {
  const a = $('ad').value.trim();
  const e = $('eposta').value.trim();
  const p = $('sifre').value;
  let s = 0;
  if (a) s = 1;
  if (a && e.includes('@')) s = 2;
  if (a && e.includes('@') && p.length >= 6) s = 3;
  ['seg1','seg2','seg3'].forEach((id, i) => {
    $(id).classList.toggle('active', i < s);
  });
}
['ad','eposta','sifre','telefon'].forEach(id => {
  $(id).addEventListener('input', updateStepBar);
});

/* ─── Toast ─────────────────────────────────────────── */
function showToast(msg, icon='bi-check-circle-fill') {
  const wrap = $('toast-wrap');
  const t = document.createElement('div');
  t.className = 'toast-item';
  t.innerHTML = `<i class="bi ${icon}"></i><span>${msg}</span>`;
  wrap.appendChild(t);
  setTimeout(() => { t.style.opacity='0'; t.style.transform='translateX(40px)'; t.style.transition='.3s'; setTimeout(()=>t.remove(), 300); }, 3500);
}

/* ─── Validation helpers ─────────────────────────────── */
function validateEmail(v) { return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(v); }
function validatePhone(v) { return /^0\d{3}\s\d{3}\s\d{2}\s\d{2}$/.test(v) || /^0\d{10}$/.test(v.replace(/\s/g,'')); }

/* ─── Form submit ────────────────────────────────────── */
$('register-form').addEventListener('submit', async function(e) {
  e.preventDefault();

  let ok = true;

  const ad = $('ad').value.trim();
  const dogum = $('dogum').value;
  const eposta = $('eposta').value.trim();
  const telefon = $('telefon').value.trim();
  const sifre = $('sifre').value;
  const sifre2 = $('sifre2').value;
  const cinsiyet = document.querySelector('input[name="cinsiyet"]:checked');
  const terms = $('terms').checked;

  /* Ad Soyad */
  const adOk = ad.length >= 3;
  setError('fc-ad', !adOk); if (!adOk) ok = false;

  /* Doğum */
  const dogumOk = dogum !== '';
  setError('fc-dogum', !dogumOk); if (!dogumOk) ok = false;

  /* E-posta */
  const epostaOk = validateEmail(eposta);
  setError('fc-eposta', !epostaOk); if (!epostaOk) ok = false;

  /* Telefon */
  const telRaw = telefon.replace(/\s/g,'');
  const telOk = telRaw.length >= 10 && telRaw.startsWith('0');
  setError('fc-telefon', !telOk); if (!telOk) ok = false;

  /* Şifre */
  const sifreOk = sifre.length >= 6;
  setError('fc-sifre', !sifreOk); if (!sifreOk) ok = false;

  /* Şifre tekrar */
  const sifre2Ok = sifre === sifre2 && sifre2 !== '';
  setError('fc-sifre2', !sifre2Ok); if (!sifre2Ok) ok = false;

  /* Cinsiyet */
  const cinsiyetOk = cinsiyet !== null;
  $('cinsiyet-error').style.display = cinsiyetOk ? 'none' : 'block';
  if (!cinsiyetOk) ok = false;

  /* Terms */
  $('terms-error').style.display = terms ? 'none' : 'block';
  if (!terms) ok = false;

  if (!ok) {
    /* Scroll to first error */
    const firstErr = document.querySelector('.has-error input, .has-error select');
    if (firstErr) firstErr.scrollIntoView({ behavior: 'smooth', block: 'center' });
    return;
  }

  /* Simulate async submission */
  const btn = $('submit-btn');
  btn.classList.add('loading');
  btn.disabled = true;

  await new Promise(r => setTimeout(r, 1800));

  btn.classList.remove('loading');
  btn.disabled = false;

  showToast('Kayıt başarıyla tamamlandı! Hoş geldiniz 🎉');
  /* In real app: redirect to dashboard */
});
