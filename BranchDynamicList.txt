    // Toplam istatistikler
    int totalBranches = Model?.Count ?? 0;
    int totalDoctors = Model?.Sum(b => b.DoctorCount) ?? 0;
    int totalAppointments = Model?.Sum(b => b.MonthlyAppointmentCount) ?? 0;
    decimal avgRating = Model != null && Model.Any() ? Math.Round(Model.Average(b => b.Rating), 1) : 0;

    // Kategori tespiti için yardımcı fonksiyon (ThemeColor veya BranchCode bazlı genişletilebilir)
    // Şu an DTO'da kategori alanı yok; ThemeColor string'inden ya da BranchCode'dan türetebilirsiniz.
    // Örnek: ilk karakter grubuna göre. Şimdilik "dahili" default.
    // Gerçek projede Category alanını DTO'ya ekleyip aşağıdaki helper'ı kullanabilirsiniz.
    string GetCategory(ResultBranchDto b)
    {
        // BranchCode veya harici mantıkla belirleyin; burada ThemeColor üzerinden örnek:
        // Cerrahi → turuncu/sarı tonları, Tanı → mavi/indigo, Dahili → diğer
        if (string.IsNullOrEmpty(b.ThemeColor)) return "dahili";
        string tc = b.ThemeColor.ToLower();
        if (tc.Contains("f59e0b") || tc.Contains("fcd34d") || tc.Contains("f97316")) return "cerrahi";
        if (tc.Contains("6366f1") || tc.Contains("a5b4fc") || tc.Contains("0ea5e9") || tc.Contains("06b6d4")) return "tani";
        return "dahili";
    }

    string GetOccupancyColor(int rate)
    {
        if (rate >= 80) return "var(--primary)";
        if (rate >= 60) return "var(--warning)";
        return "var(--success)";
    }

    string GetStatusClass(bool status) => status ? "on" : "off";
    string GetStatusLabel(bool status) => status ? "Aktif" : "Pasif";
}

    <script>
        const BC = (() => {
            let _cat     = 'all';
            let _delName = '';
            let _delId   = '';
            let _delCard = null;

            /* ─ View ─ */
            function setView(v) {
                document.getElementById('bc-grid').classList.toggle('list-mode', v === 'list');
                document.getElementById('bc-vg').classList.toggle('active', v === 'grid');
                document.getElementById('bc-vl').classList.toggle('active', v === 'list');
                if (v === 'list') _closeAllExpand();
            }

            /* ─ Tab ─ */
            function setTab(el) {
                document.querySelectorAll('.bc-tab').forEach(t => t.classList.remove('active'));
                el.classList.add('active');
                _cat = el.dataset.cat;
                filter();
            }

            /* ─ Filter ─ */
            function filter() {
                const q = (document.getElementById('bc-search').value || '').toLowerCase().trim();
                let n = 0;
                document.querySelectorAll('#bc-grid .bc-card').forEach(c => {
                    const show = (_cat === 'all' || c.dataset.cat === _cat)
                              && (c.dataset.name || '').includes(q);
                    c.style.display = show ? '' : 'none';
                    if (show) n++;
                });

                // Stat sayacını güncelle
                const cnt = document.getElementById('bc-count');
                if (cnt) cnt.textContent = n;

                // Empty state
                let empty = document.getElementById('bc-empty-dyn');
                if (n === 0 && !empty) {
                    empty = document.createElement('div');
                    empty.id = 'bc-empty-dyn';
                    empty.className = 'bc-empty';
                    empty.innerHTML = '<i class="bi bi-search"></i>'
                        + '<div class="bc-empty-title">Sonuç bulunamadı</div>'
                        + '<div style="font-size:13px">Farklı arama terimi veya kategori deneyin.</div>';
                    document.getElementById('bc-grid').appendChild(empty);
                } else if (n > 0 && empty) {
                    empty.remove();
                }
            }

            /* ─ Docs expand ─ */
            function toggleDocs(i, branchId, btn) {
                const panel = document.getElementById('bc-exp-' + i);
                const open  = panel.classList.contains('open');
                _closeAllExpand();
                if (!open) {
                    panel.classList.add('open');
                    btn.innerHTML = '<i class="bi bi-chevron-up"></i> Gizle';
                    // Doktor listesi zaten yüklenmediyse fetch et
                    if (!panel.dataset.loaded) {
                        _loadDoctors(panel, branchId);
                    }
                }
            }

            function _loadDoctors(panel, branchId) {
                // Gerçek endpoint'inize göre URL'yi güncelleyin
                // Örnek: /Admin/Branch/GetDoctors?branchId=...
                fetch(`/Admin/Branch/GetDoctors?branchId=${branchId}`)
                    .then(r => {
                        if (!r.ok) throw new Error('HTTP ' + r.status);
                        return r.json();
                    })
                    .then(doctors => {
                        panel.dataset.loaded = 'true';
                        if (!doctors || doctors.length === 0) {
                            panel.innerHTML = '<div class="bc-expand-title">Doktorlar</div>'
                                + '<div style="text-align:center;padding:16px 0;color:var(--text-muted);font-size:13px">'
                                + 'Bu branşa kayıtlı doktor bulunamadı.</div>';
                            return;
                        }

                        let html = '<div class="bc-expand-title">Doktorlar</div>';
                        const colors = ['#3B82F6','#10B981','#F59E0B','#7C3AED','#06B6D4','#6366F1'];
                        doctors.forEach((d, idx) => {
                            const initials = (d.fullName || 'Dr')
                                .split(' ').filter(Boolean)
                                .slice(0, 2).map(w => w[0].toUpperCase()).join('');
                            const color    = colors[idx % colors.length];
                            const avail    = d.isAvailable !== false;
                            const badgeCls = avail ? 'on' : 'off';
                            const badgeTxt = avail ? 'Müsait' : 'Meşgul';
                            const btnHtml  = avail
                                ? `<button class="bc-appt-btn" onclick="BC.book('${d.fullName}')"><i class="bi bi-calendar-plus"></i> Randevu</button>`
                                : `<button class="bc-appt-btn" disabled><i class="bi bi-clock"></i> Dolu</button>`;
                            html += `<div class="bc-doc-row">
                                <div class="bc-doc-av" style="background:${color}">${initials}</div>
                                <div>
                                    <div class="bc-doc-name">${d.fullName || ''}</div>
                                    <div class="bc-doc-spec">${d.specialty || ''}</div>
                                </div>
                                <span class="bc-doc-badge ${badgeCls}">${badgeTxt}</span>
                                ${btnHtml}
                            </div>`;
                        });

                        panel.innerHTML = html;
                    })
                    .catch(() => {
                        panel.innerHTML = '<div class="bc-expand-title">Doktorlar</div>'
                            + '<div style="text-align:center;padding:16px 0;color:var(--danger);font-size:13px">'
                            + '<i class="bi bi-exclamation-circle" style="display:block;margin-bottom:6px"></i>'
                            + 'Doktorlar yüklenemedi.</div>';
                    });
            }

            function _closeAllExpand() {
                document.querySelectorAll('.bc-expand.open').forEach(p => p.classList.remove('open'));
                document.querySelectorAll('.bc-btn.primary').forEach(b => {
                    if (b.textContent.includes('Gizle')) {
                        b.innerHTML = '<i class="bi bi-people"></i> Doktorlar';
                    }
                });
            }

            /* ─ Appointment ─ */
            function book(doc) {
                Shell.toast(`${doc} için randevu talebi oluşturuldu!`, 'success', 'bi-calendar-check');
            }

            /* ─ Modal helpers ─ */
            function open(id) {
                document.getElementById(id).classList.add('show');
                document.body.style.overflow = 'hidden';
            }
            function close(id) {
                document.getElementById(id).classList.remove('show');
                document.body.style.overflow = '';
            }

            /* ─ Add ─ */
            function openAdd() { open('bc-m-add'); }

            /* ─ Edit ─ */
            function openEdit(d) {
                document.getElementById('bc-edit-id').value   = d.id   || '';
                document.getElementById('bc-edit-name').value = d.name || '';
                document.getElementById('bc-edit-desc').value = d.desc || '';
                document.getElementById('bc-edit-docs').value = d.docs || '';
                document.getElementById('bc-edit-code').textContent = 'Kod: ' + (d.code || '');
                document.getElementById('bc-edit-cat').value  = d.cat  || 'dahili';
                open('bc-m-edit');
            }

            function saveEdit() {
                const n  = document.getElementById('bc-edit-name').value.trim();
                const id = document.getElementById('bc-edit-id').value;
                if (!n) {
                    Shell.toast('Branş adı boş olamaz!', 'danger', 'bi-exclamation-circle');
                    return;
                }

                // Gerçek API çağrısı için:
                // fetch('/Admin/Branch/Update', { method:'POST', headers:{'Content-Type':'application/json'},
                //   body: JSON.stringify({ BranchId: id, BranchName: n, ... }) })
                // .then(r => r.json()).then(res => { if(res.success) { close('bc-m-edit'); location.reload(); } });

                close('bc-m-edit');
                Shell.toast(`"${n}" branşı güncellendi.`, 'success', 'bi-check-circle');
            }

            /* ─ Delete ─ */
            function confirmDelete(name, id, btn) {
                _delName = name;
                _delId   = id;
                _delCard = btn.closest('.bc-card');
                document.getElementById('bc-del-name').textContent = name;
                open('bc-m-delete');
            }

            function execDelete() {
                close('bc-m-delete');

                // Gerçek API çağrısı için:
                // fetch(`/Admin/Branch/Delete?id=${_delId}`, { method:'DELETE' })
                // .then(r => r.json()).then(res => { if(res.success) { _removeCard(); } });

                _removeCard();
                Shell.toast(`"${_delName}" branşı silindi.`, 'danger', 'bi-trash3');
            }

            function _removeCard() {
                if (_delCard) {
                    _delCard.style.transition = 'opacity .3s ease, transform .3s ease';
                    _delCard.style.opacity    = '0';
                    _delCard.style.transform  = 'scale(.94)';
                    setTimeout(() => { _delCard?.remove(); _delCard = null; filter(); }, 320);
                }
            }

            /* ─ Export ─ */
            function exportList() {
                // Gerçek export endpoint'i:
                // window.location.href = '/Admin/Branch/Export';
                Shell.toast('Branş listesi Excel olarak hazırlanıyor...', 'info', 'bi-file-earmark-spreadsheet');
            }

            /* ─ ESC ─ */
            document.addEventListener('keydown', e => {
                if (e.key === 'Escape')
                    document.querySelectorAll('.bc-overlay.show').forEach(m => close(m.id));
            });

            /* ─ Topbar başlığını güncelle ─ */
            if (typeof Shell !== 'undefined' && Shell.setPage)
                Shell.setPage('Branşlar', 'Admin Paneli / Klinik / Branşlar');

            return {
                setView, setTab, filter, toggleDocs,
                book, openAdd, openEdit, saveEdit,
                confirmDelete, execDelete, exportList, close
            };
        })();
    </script>
