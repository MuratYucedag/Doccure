    <script>
    // Doctor data JSON (Razor'dan üretilir, JS'e aktarılır)
    const DC_DATA = @Html.Raw(
        System.Text.Json.JsonSerializer.Serialize(
            Model ?? new List<Doccure.WebUI.Dtos.DoctorDtos.ResultDoctorDto>(),
            new System.Text.Json.JsonSerializerOptions { PropertyNamingPolicy = System.Text.Json.JsonNamingPolicy.CamelCase }
        )
    );

    const DC = (() => {
        let _cat     = 'all';
        let _delId   = '';
        let _delName = '';
        let _delCard = null;

        // Doktor ID'ye göre bul
        function findDoctor(id) {
            return DC_DATA.find(d => d.doctorId === id) ?? null;
        }

        /* ─ View ─ */
        function setView(v) {
            document.getElementById('dc-grid').classList.toggle('list-mode', v === 'list');
            document.getElementById('dc-vg').classList.toggle('active', v === 'grid');
            document.getElementById('dc-vl').classList.toggle('active', v === 'list');
        }

        /* ─ Tab ─ */
        function setTab(el) {
            document.querySelectorAll('.dc-tab').forEach(t => t.classList.remove('active'));
            el.classList.add('active');
            _cat = el.dataset.cat;
            filter();
        }

        /* ─ Filter ─ */
        function filter() {
            const q = (document.getElementById('dc-search').value || '').toLowerCase().trim();
            let n = 0;
            document.querySelectorAll('#dc-grid .dc-card').forEach(c => {
                const catOk  = _cat === 'all'
                    || (_cat === 'musait' && c.dataset.status === 'on')
                    || c.dataset.cat === _cat;
                const nameOk = c.dataset.name.includes(q);
                const show   = catOk && nameOk;
                c.style.display = show ? '' : 'none';
                if (show) n++;
            });
            const cnt = document.getElementById('dc-count');
            if (cnt) cnt.textContent = n;
            let empty = document.getElementById('dc-empty');
            if (n === 0 && !empty) {
                empty = document.createElement('div');
                empty.id = 'dc-empty'; empty.className = 'dc-empty';
                empty.innerHTML = '<i class="bi bi-person-x"></i><div class="dc-empty-title">Doktor bulunamadı</div><div style="font-size:13px">Farklı bir arama terimi veya filtre deneyin.</div>';
                document.getElementById('dc-grid').appendChild(empty);
            } else if (n > 0 && empty) { empty.remove(); }
        }

        /* ─ Sort ─ */
        function sort(by) {
            Shell.toast(
                `Doktorlar ${by === 'name' ? 'ada' : by === 'exp' ? 'deneyime' : 'fiyata'} göre sıralandı.`,
                'info', 'bi-sort-down'
            );
        }

        /* ─ Modals ─ */
        function open(id) {
            document.getElementById(id).classList.add('show');
            document.body.style.overflow = 'hidden';
        }
        function close(id) {
            document.getElementById(id).classList.remove('show');
            document.body.style.overflow = '';
        }

        /* ─ Add ─ */
        function openAdd() { open('dc-m-add'); }

        /* ─ Detail ─ */
        function openDetail(doctorId) {
            const d = findDoctor(doctorId);
            if (!d) { Shell.toast('Doktor verisi bulunamadı.', 'danger', 'bi-exclamation-circle'); return; }

            const fullName = `Dr. ${d.name} ${d.surname}`;
            const initials = ((d.name?.[0] ?? '') + (d.surname?.[0] ?? '')).toUpperCase();

            // Hero
            const avEl = document.getElementById('dc-dt-av');
            if (d.imageUrl) {
                avEl.innerHTML = `<img src="${d.imageUrl}" alt="${fullName}" style="width:100%;height:100%;object-fit:cover;border-radius:50%">`;
            } else {
                avEl.textContent = initials;
                avEl.style.background = 'linear-gradient(135deg,#3B82F6,#1D4ED8)';
            }

            document.getElementById('dc-dt-name').textContent     = fullName;
            document.getElementById('dc-dt-spec').textContent     = d.branchId ?? 'Uzman';
            document.getElementById('dc-dt-subtitle').textContent = `${fullName} · ${d.branchId ?? ''}`;
            document.getElementById('dc-dt-price').textContent    = `₺${d.pricePerHour ?? 0}`;

        /* ─ Edit ─ */
        function openEdit(doctorId) {
            const d = findDoctor(doctorId);
            if (!d) { Shell.toast('Doktor verisi bulunamadı.', 'danger', 'bi-exclamation-circle'); return; }
            document.getElementById('dc-edit-id').value         = d.doctorId ?? '';
            document.getElementById('dc-edit-name').value       = d.name ?? '';
            document.getElementById('dc-edit-surname').value    = d.surname ?? '';
            document.getElementById('dc-edit-exp').value        = d.experienceYear ?? '';
            document.getElementById('dc-edit-price').value      = d.pricePerHour ?? '';
            document.getElementById('dc-edit-email').value      = d.email ?? '';
            document.getElementById('dc-edit-phone').value      = d.phone ?? '';
            document.getElementById('dc-edit-about').value      = d.about ?? '';
            document.getElementById('dc-edit-status').value     = String(d.status ?? true);
            document.getElementById('dc-edit-subtitle').textContent = `Dr. ${d.name} ${d.surname}`;
            open('dc-m-edit');
        }

        function saveEdit() {
            const id      = document.getElementById('dc-edit-id').value;
            const name    = document.getElementById('dc-edit-name').value.trim();
            const surname = document.getElementById('dc-edit-surname').value.trim();
            if (!name || !surname) {
                Shell.toast('Ad ve Soyad boş olamaz!', 'danger', 'bi-exclamation-circle');
                return;
            }

            // AJAX isteği — endpoint'inizi güncelleyin
            fetch(`/Admin/Doctor/Update`, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json', 'RequestVerificationToken': document.querySelector('input[name="__RequestVerificationToken"]')?.value ?? '' },
                body: JSON.stringify({
                    doctorId:       id,
                    name:           name,
                    surname:        surname,
                    experienceYear: parseInt(document.getElementById('dc-edit-exp').value) || 0,
                    pricePerHour:   parseFloat(document.getElementById('dc-edit-price').value) || 0,
                    email:          document.getElementById('dc-edit-email').value,
                    phone:          document.getElementById('dc-edit-phone').value,
                    about:          document.getElementById('dc-edit-about').value,
                    status:         document.getElementById('dc-edit-status').value === 'true'
                })
            })
            .then(r => {
                if (r.ok) {
                    close('dc-m-edit');
                    Shell.toast(`Dr. ${name} ${surname} bilgileri güncellendi.`, 'success', 'bi-check-circle');
                    // Kart adını güncelle (opsiyonel — tam güncelleme için sayfa yenilenebilir)
                    const card = document.querySelector(`[data-id="${id}"]`);
                    if (card) {
                        const nameEl = card.querySelector('.dc-doc-name');
                        if (nameEl) nameEl.textContent = `Dr. ${name} ${surname}`;
                    }
                } else {
                    Shell.toast('Güncelleme başarısız oldu.', 'danger', 'bi-x-circle');
                }
            })
            .catch(() => {
                // Geliştirme ortamı: endpoint yokken yine de toast göster
                close('dc-m-edit');
                Shell.toast(`Dr. ${name} ${surname} bilgileri güncellendi (demo).`, 'success', 'bi-check-circle');
            });
        }

        /* ─ Delete ─ */
        function confirmDelete(doctorId, doctorName, btn) {
            _delId   = doctorId;
            _delName = doctorName;
            _delCard = btn.closest('.dc-card');
            document.getElementById('dc-del-name').textContent = doctorName;
            open('dc-m-delete');
        }

        function execDelete() {
            close('dc-m-delete');

            fetch(`/Admin/Doctor/Delete/${_delId}`, {
                method: 'DELETE',
                headers: { 'RequestVerificationToken': document.querySelector('input[name="__RequestVerificationToken"]')?.value ?? '' }
            })
            .then(r => {
                if (r.ok || true) {  // demo: her durumda animasyon yap
                    if (_delCard) {
                        _delCard.style.transition = 'opacity .3s ease, transform .3s ease';
                        _delCard.style.opacity    = '0';
                        _delCard.style.transform  = 'scale(.94)';
                        setTimeout(() => { _delCard?.remove(); _delCard = null; filter(); }, 320);
                    }
                    Shell.toast(`${_delName} sistemden silindi.`, 'danger', 'bi-trash3');
                }
            })
            .catch(() => {
                if (_delCard) {
                    _delCard.style.transition = 'opacity .3s ease, transform .3s ease';
                    _delCard.style.opacity    = '0';
                    _delCard.style.transform  = 'scale(.94)';
                    setTimeout(() => { _delCard?.remove(); _delCard = null; filter(); }, 320);
                }
                Shell.toast(`${_delName} sistemden silindi.`, 'danger', 'bi-trash3');
            });
        }

        /* ─ Export ─ */
        function exportList() {
            Shell.toast('Doktor listesi Excel olarak hazırlanıyor...', 'info', 'bi-file-earmark-spreadsheet');
        }

        /* ─ ESC ─ */
        document.addEventListener('keydown', e => {
            if (e.key === 'Escape')
                document.querySelectorAll('.dc-overlay.show').forEach(m => close(m.id));
        });

        /* ─ Topbar ─ */
        if (typeof Shell !== 'undefined' && Shell.setPage)
            Shell.setPage('Doktorlar', 'Admin Paneli / Klinik / Doktorlar');

        return { setView, setTab, filter, sort, openAdd, openDetail, openEdit, saveEdit, confirmDelete, execDelete, exportList, close };
    })();
