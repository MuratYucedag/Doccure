        document.getElementById('da-prev-btn').style.display = s === 0 ? 'none' : '';
        document.getElementById('da-next-btn').style.display = s === TOTAL - 1 ? 'none' : '';
        document.getElementById('da-submit-btn').style.display = s === TOTAL - 1 ? '' : 'none';
        document.getElementById('da-step-indicator').textContent = `Adım ${s + 1} / ${TOTAL}`;
        window.scrollTo({ top: 0, behavior: 'smooth' });

            el.classList.toggle('has-error', err);
            if (err) ok = false;

            // Re-number
            listEl.querySelectorAll('.da-item').forEach((el, i) => {
                el.querySelector('.da-item-num').textContent = i + 1;
            });
            updatePreview();
        }, 260);
    }

    function refreshSummary(type, idx) {
        const cfg  = ITEM_CONFIG[type];
        const list = document.getElementById(cfg.listId);
        const item = list.querySelector(`[data-idx="${idx}"]`);
        if (!item) return;
        const vals = {};
        cfg.fields.forEach(f => {
            // name formatı: "Educations[0].SchoolName"
            const el = item.querySelector(`[name="${cfg.prefix}[${idx}].${f.name}"]`);
            if (el) vals[f.name] = el.value.trim();
        });
        const sumText = item.querySelector('.da-item-sum-text');
        if (sumText) sumText.textContent = cfg.summary(vals);
    }

    function _refreshCount(type) {
        const cfg = ITEM_CONFIG[type];
        const listEl = document.getElementById(cfg.listId);
        const n = listEl.querySelectorAll('.da-item').length;
        document.getElementById(cfg.countId).textContent = n;
        if (cfg.previewId) document.getElementById(cfg.previewId).textContent = n;
    }

    // ── Tag Input ──
    function initTagInput(wrapId, rawId, hiddenId, namePfx, arr) {
        const raw = document.getElementById(rawId);
        raw.addEventListener('keydown', e => {
            if (e.key === 'Enter' || e.key === ',') {
                e.preventDefault();
                const val = raw.value.replace(',', '').trim();
                if (val) addTag(wrapId, hiddenId, namePfx, arr, val, raw);
                raw.value = '';
            }
            if (e.key === 'Backspace' && !raw.value && arr.length) {
                removeTag(wrapId, hiddenId, arr, arr[arr.length - 1]);
            }
        });
    }

    function addTag(wrapId, hiddenId, namePfx, arr, val, rawEl) {
        if (arr.includes(val)) return;
        arr.push(val);
        const wrap    = document.getElementById(wrapId);
        const hiddenW = document.getElementById(hiddenId);
        const pill = document.createElement('span');
        pill.className = 'da-tag-pill';
        pill.dataset.val = val;
        pill.innerHTML = `${escHtml(val)}<button class="da-tag-pill-del" type="button" onclick="DA._removeTagByVal('${wrapId}','${hiddenId}',${JSON.stringify(arr)},'${val}')"><i class="bi bi-x-lg"></i></button>`;
        wrap.insertBefore(pill, rawEl);
        // MVC binding: Services[0], Services[1], ...  (indexed, not [])
        _rebuildHiddenInputs(hiddenW, namePfx, arr);
        updatePreview();
        document.getElementById('prev-svcs').textContent = _svcTags.length;
    }

    // Tüm hidden input'ları sıfırdan yeniden oluşturur — index tutarlılığı için
    function _rebuildHiddenInputs(container, namePfx, arr) {
        container.innerHTML = '';
        arr.forEach((v, i) => {
            const inp = document.createElement('input');
            inp.type  = 'hidden';
            inp.name  = `${namePfx}[${i}]`;   // Specializations[0], Services[1], ...
            inp.value = v;
            container.appendChild(inp);
        });
    }

    function _removeTagByVal(wrapId, hiddenId, arr, val) {
        removeTag(wrapId, hiddenId, arr, val);
    }

    function removeTag(wrapId, hiddenId, arr, val) {
        const i = arr.indexOf(val);
        if (i > -1) arr.splice(i, 1);
        const wrap    = document.getElementById(wrapId);
        const hiddenW = document.getElementById(hiddenId);
        wrap.querySelector(`[data-val="${CSS.escape(val)}"]`)?.remove();
        // Tüm hidden input'ları yeniden oluştur (index'ler kaymasın diye)
        const namePfx = hiddenW === document.getElementById('spec-hidden-inputs') ? 'Specializations' : 'Services';
        _rebuildHiddenInputs(hiddenW, namePfx, arr);
        updatePreview();
        document.getElementById('prev-svcs').textContent = _svcTags.length;
    }

    function escHtml(s) {
        return s.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
    }

    // ── Before Submit ──
    function beforeSubmit() {
        // Renumber hidden inputs so indices are sequential
        ['edu','exp','loc','award'].forEach(type => {
            const list = document.getElementById(ITEM_CONFIG[type].listId);
            list.querySelectorAll('.da-item').forEach((item, i) => {
                item.querySelectorAll('.da-item-field').forEach(field => {
                    const oldName = field.name;
                    field.name = oldName.replace(/\[\d+\]/, `[${i}]`);
                });
            });
        });
        Shell?.toast('Doktor kaydediliyor...', 'info', 'bi-arrow-repeat');
    }

    // ── Init ──
    (function init() {
        initTagInput('spec-wrap','spec-raw','spec-hidden-inputs','Specializations', _specTags);
        initTagInput('svc-wrap', 'svc-raw', 'svc-hidden-inputs', 'Services',       _svcTags);

        // Remove has-error on input
        document.querySelectorAll('.da-form-control').forEach(el => {
            el.addEventListener('input', () => el.classList.remove('has-error'));
        });

        if (typeof Shell !== 'undefined' && Shell.setPage)
            Shell.setPage('Yeni Doktor', 'Admin Paneli / Klinik / Doktorlar / Ekle');
    })();

    return {
        goStep, nextStep, prevStep,
        addItem, toggleItem, removeItem, refreshSummary,
        updatePreview, previewImg,
        _removeTagByVal, beforeSubmit,
    };
})();
</script>
