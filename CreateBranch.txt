    <script>
        const CB = (() => {

            /* ─ Preview sync ─ */
            function syncPreview() {
                const name  = document.getElementById('cb-name').value        || 'Branş Adı';
                const code  = document.getElementById('cb-code').value        || 'BRN-000';
                const desc  = document.getElementById('cb-desc').value        || 'Branş açıklaması burada görünecek...';
                const docs  = document.getElementById('cb-doctors').value     || '0';
                const appt  = document.getElementById('cb-appointments').value|| '0';
                const beds  = document.getElementById('cb-beds').value        || '0';
                const rat   = parseFloat(document.getElementById('cb-rating').value || 0).toFixed(1);
                const occ   = parseInt(document.getElementById('cb-occupancy').value || 0);

                document.getElementById('prev-name').textContent    = name;
                document.getElementById('prev-code').textContent    = '#' + code;
                document.getElementById('prev-desc').textContent    = desc;
                document.getElementById('prev-docs').textContent    = docs;
                document.getElementById('prev-appt').textContent    = appt;
                document.getElementById('prev-beds').textContent    = beds;
                document.getElementById('prev-rating').textContent  = rat;

                const clamp = Math.min(100, Math.max(0, occ));
                const occBar = document.getElementById('prev-occ-bar');
                const occPct = document.getElementById('prev-occ-pct');
                const occColor = clamp >= 80 ? 'var(--primary)' : clamp >= 60 ? 'var(--warning)' : 'var(--success)';
                occBar.style.width      = clamp + '%';
                occBar.style.background = occColor;
                occPct.textContent      = clamp + '%';
                occPct.style.color      = occColor;
            }

            /* ─ Status toggle ─ */
            function syncStatus() {
                const chk = document.getElementById('cb-status');
                const lbl = document.getElementById('cb-status-lbl');
                if (chk.checked) {
                    lbl.textContent = 'Aktif';
                    lbl.className   = 'cb-toggle-state on';
                } else {
                    lbl.textContent = 'Pasif';
                    lbl.className   = 'cb-toggle-state off';
                }
            }

            /* ─ Emoji picker ─ */
            function toggleEmojiPicker() {
                document.getElementById('cb-emoji-grid').classList.toggle('open');
            }

            function selectEmoji(emoji) {
                document.getElementById('cb-imageurl').value  = emoji;
                document.getElementById('cb-emoji-preview').textContent = emoji;
                document.getElementById('prev-icon').textContent        = emoji;

                // Seçili state
                document.querySelectorAll('.cb-emoji-btn').forEach(b => {
                    b.classList.toggle('selected', b.textContent.trim() === emoji);
                });

                document.getElementById('cb-emoji-grid').classList.remove('open');
            }

            function onEmojiInput(val) {
                const v = val.trim() || '🏥';
                document.getElementById('cb-emoji-preview').textContent = v;
                document.getElementById('prev-icon').textContent         = v;
            }

            /* ─ Color picker ─ */
            function selectColor(gradVal, el) {
                document.querySelectorAll('.cb-color-swatch').forEach(s => s.classList.remove('selected'));
                el.classList.add('selected');
                document.getElementById('cb-themecolor').value  = gradVal;
                document.getElementById('cb-color-bar').style.background = gradVal;
                document.getElementById('prev-accent').style.background  = gradVal;
            }

            /* ─ Number stepper ─ */
            function stepNum(id, step) {
                const el  = document.getElementById(id);
                const min = parseFloat(el.min ?? '-Infinity');
                const max = parseFloat(el.max ?? 'Infinity');
                let val   = parseFloat(el.value || 0) + step;
                val = Math.min(max, Math.max(min, Math.round(val * 10) / 10));
                el.value  = val;
                syncPreview();
            }

            /* ─ Validate ─ */
            function validate() {
                let ok = true;
                const required = [
                    { id: 'cb-name', msg: 'Branş adı zorunludur.' },
                    { id: 'cb-code', msg: 'Branş kodu zorunludur.' }
                ];
                required.forEach(r => {
                    const el = document.getElementById(r.id);
                    if (!el.value.trim()) {
                        el.classList.add('is-invalid');
                        ok = false;
                    } else {
                        el.classList.remove('is-invalid');
                    }
                });
                return ok;
            }

            /* ─ Submit ─ */
            function submit() {
                if (!validate()) {
                    Shell.toast('Lütfen zorunlu alanları doldurun.', 'danger', 'bi-exclamation-circle');
                    return;
                }

                const btn     = document.getElementById('cb-save-btn');
                const spinner = document.getElementById('cb-spinner');
                const icon    = document.getElementById('cb-save-icon');
                const lbl     = document.getElementById('cb-save-lbl');

                btn.disabled          = true;
                spinner.style.display = 'block';
                icon.style.display    = 'none';
                lbl.textContent       = 'Kaydediliyor...';

                document.getElementById('cb-form').submit();
            }

            /* ─ Reset ─ */
            function resetForm() {
                document.getElementById('cb-form').reset();
                document.getElementById('cb-emoji-preview').textContent = '🏥';
                document.getElementById('prev-icon').textContent         = '🏥';
                document.querySelectorAll('.cb-color-swatch').forEach((s, i) => s.classList.toggle('selected', i === 0));
                const defaultGrad = 'linear-gradient(90deg,#3B82F6,#60A5FA)';
                document.getElementById('cb-themecolor').value           = defaultGrad;
                document.getElementById('cb-color-bar').style.background = defaultGrad;
                document.getElementById('prev-accent').style.background  = defaultGrad;
                document.querySelectorAll('.cb-form-control.is-invalid').forEach(el => el.classList.remove('is-invalid'));
                syncPreview();
                syncStatus();
                Shell.toast('Form sıfırlandı.', 'info', 'bi-arrow-counterclockwise');
            }

            /* ─ Dışarı tıklayınca emoji picker kapat ─ */
            document.addEventListener('click', e => {
                const grid    = document.getElementById('cb-emoji-grid');
                const preview = document.getElementById('cb-emoji-preview');
                if (!grid.contains(e.target) && e.target !== preview) {
                    grid.classList.remove('open');
                }
            });

            /* ─ Topbar ─ */
            if (typeof Shell !== 'undefined' && Shell.setPage)
                Shell.setPage('Yeni Branş', 'Admin Paneli / Klinik / Branşlar / Yeni');

            /* ─ İlk render ─ */
            syncPreview();
            // Accent bar başlangıç
            document.getElementById('prev-accent').style.background = 'linear-gradient(90deg,#3B82F6,#60A5FA)';

            return { syncPreview, syncStatus, toggleEmojiPicker, selectEmoji, onEmojiInput, selectColor, stepNum, submit, resetForm };
        })();
    </script>
