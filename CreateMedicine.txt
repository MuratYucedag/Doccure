        <div class="mc-action-bar">
            <div class="mc-action-left">
                <a href="/Admin/Medicine/List" class="mc-btn mc-btn-outline">
                    <i class="bi bi-arrow-left"></i> İptal
                </a>
                <button type="button" class="mc-btn mc-btn-outline" onclick="MC.resetForm()">
                    <i class="bi bi-arrow-counterclockwise"></i> Sıfırla
                </button>
            </div>
            <div class="mc-action-right">
                <button type="submit" class="mc-btn mc-btn-success" id="mc-submit-btn" onclick="return MC.validate()">
                    <i class="bi bi-check2-circle"></i> İlacı Kaydet
                </button>
            </div>
        </div>

    </form>
</div><!-- /.mc-page -->

<script>
const MC = (() => {

    /* ── Önizleme güncelle ── */
    function updatePreview() {
        const name     = v('MedicineName');
        const barcode  = v('Barcode');
        const stock    = parseInt(v('Stock'))             || 0;
        const critical = parseInt(v('CriticalStockLevel')) || 0;
        const price    = parseFloat(v('UnitPrice'))       || 0;
        const expDate  = v('ExpirationDate');
        const isActive = document.getElementById('Status')?.checked;

        // İsim
        set('prev-name',    name    || 'İlaç adı girilmedi');
        set('prev-barcode', barcode ? `Barkod: ${barcode}` : 'Barkod girilmedi');

        // Stats
        set('prev-stock',    stock    > 0  ? stock + ' adet'                    : '—');
        set('prev-critical', critical > 0  ? critical + ' adet'                 : '—');
        set('prev-price',    price    > 0  ? '₺' + price.toFixed(2)             : '—');

        // SKT
        if (expDate) {
            const d = new Date(expDate);
            set('prev-expiry', d.toLocaleDateString('tr-TR', {day:'2-digit',month:'2-digit',year:'numeric'}));
        } else {
            set('prev-expiry', '—');
        }

        // Durum label
        const lbl = document.getElementById('mc-status-label');
        if (lbl) lbl.textContent = isActive ? 'Aktif' : 'Pasif';

        // Checklist
        chk('chk-name',     !!name);
        chk('chk-stock',    stock > 0);
        chk('chk-critical', critical > 0);
        chk('chk-price',    price > 0);
        chk('chk-date',     !!expDate);
    }

    /* ── Stok durumu bar ── */
    function updateStockBar() {
        const stock    = parseInt(v('Stock'))             || 0;
        const critical = parseInt(v('CriticalStockLevel')) || 0;
        const bar      = document.getElementById('mc-stock-bar');
        const hint     = document.getElementById('mc-stock-hint');
        if (!bar || !hint) return;

        if (critical <= 0) {
            bar.style.width = '0%';
            hint.textContent = 'Kritik seviye girilince stok durumu gösterilir.';
            hint.style.color = 'var(--m-ink-3)';
            return;
        }

        const max = critical * 4;
        const pct = Math.min(100, Math.round((stock / max) * 100));
        bar.style.width = pct + '%';

        if (stock === 0) {
            bar.style.background = 'var(--m-red)';
            hint.innerHTML = '<i class="bi bi-x-circle-fill"></i> Stok tükendi!';
            hint.style.color = 'var(--m-red)';
        } else if (stock <= critical) {
            bar.style.background = 'var(--m-red)';
            hint.innerHTML = '<i class="bi bi-exclamation-triangle-fill"></i> Kritik seviyede! Hemen sipariş verin.';
            hint.style.color = 'var(--m-red)';
        } else if (stock <= critical * 2) {
            bar.style.background = 'var(--m-amber)';
            hint.innerHTML = '<i class="bi bi-exclamation-circle"></i> Stok azalıyor, yakında sipariş gerekebilir.';
            hint.style.color = 'var(--m-amber)';
        } else {
            bar.style.background = 'var(--m-green)';
            hint.innerHTML = '<i class="bi bi-check-circle"></i> Stok seviyesi yeterli.';
            hint.style.color = 'var(--m-green)';
        }
    }

    /* ── SKT ipucu ── */
    function updateDateHint() {
        const val  = v('ExpirationDate');
        const hint = document.getElementById('mc-date-hint');
        if (!hint) return;
        if (!val) { hint.classList.remove('show'); return; }

        const days = (new Date(val) - new Date()) / 86400000;
        hint.classList.add('show');
        hint.className = hint.className.replace(/\b(ok|warn|danger)\b/g, '');

        if (days < 0) {
            hint.classList.add('danger');
            hint.innerHTML = '<i class="bi bi-x-circle-fill"></i> Bu tarihin süresi zaten geçmiş!';
        } else if (days <= 30) {
            hint.classList.add('danger');
            hint.innerHTML = `<i class="bi bi-exclamation-triangle-fill"></i> Sadece ${Math.floor(days)} gün geçerli — çok kısa!`;
        } else if (days <= 90) {
            hint.classList.add('warn');
            hint.innerHTML = `<i class="bi bi-exclamation-circle"></i> ${Math.floor(days)} gün geçerli — yakın vadeli.`;
        } else {
            hint.classList.add('ok');
            hint.innerHTML = `<i class="bi bi-check-circle"></i> ${Math.floor(days)} gün geçerli.`;
        }
    }

    /* ── Validasyon ── */
    function validate() {
        let ok = true;

        const checks = [
            ['MedicineName', 'err-name',     val => !!val.trim()],
            ['Stock',        'err-stock',    val => parseInt(val) >= 0 && val !== ''],
            ['CriticalStockLevel', 'err-critical', val => parseInt(val) > 0],
            ['UnitPrice',    'err-price',    val => parseFloat(val) > 0],
            ['ExpirationDate','err-date',    val => !!val],
        ];

        checks.forEach(([id, errId, test]) => {
            const el  = document.getElementById(id);
            const err = document.getElementById(errId);
            if (!el || !err) return;
            const pass = test(el.value);
            el.classList.toggle('has-error',   !pass);
            el.classList.toggle('has-success',  pass);
            err.classList.toggle('show', !pass);
            if (!pass) ok = false;
        });

        if (!ok) {
            // İlk hatalı alana scroll
            const firstErr = document.querySelector('.mc-input.has-error');
            if (firstErr) firstErr.scrollIntoView({ behavior:'smooth', block:'center' });
            Shell?.toast('Lütfen zorunlu alanları doldurun.', 'danger', 'bi-exclamation-circle');
        }
        return ok;
    }

    /* ── Formu sıfırla ── */
    function resetForm() {
        document.getElementById('mc-form')?.reset();
        document.querySelectorAll('.mc-input').forEach(el => {
            el.classList.remove('has-error', 'has-success');
        });
        document.querySelectorAll('.mc-err').forEach(el => el.classList.remove('show'));
        document.getElementById('mc-date-hint')?.classList.remove('show');
        updatePreview();
        updateStockBar();
        Shell?.toast('Form sıfırlandı.', 'info', 'bi-arrow-counterclockwise');
    }

    /* ── Input temizleme: has-error ── */
    document.querySelectorAll('.mc-input').forEach(el => {
        el.addEventListener('input', () => {
            el.classList.remove('has-error');
            const errId = 'err-' + el.id.toLowerCase().replace('medicinename','name')
                                                       .replace('criticalstocklevel','critical')
                                                       .replace('unitprice','price')
                                                       .replace('expirationdate','date');
            document.getElementById(errId)?.classList.remove('show');
        });
    });

    /* ── Shell setPage ── */
    if (typeof Shell !== 'undefined' && Shell.setPage)
        Shell.setPage('Yeni İlaç', 'Admin Paneli / Depo / İlaç Yönetimi / Ekle');

    /* ── Yardımcılar ── */
    function v(id)      { return document.getElementById(id)?.value ?? ''; }
    function set(id, t) { const el = document.getElementById(id); if (el) el.textContent = t; }
    function chk(id, done) {
        const el = document.getElementById(id);
        if (el) el.classList.toggle('done', done);
    }

    return { updatePreview, updateStockBar, updateDateHint, validate, resetForm };
})();
</script>

</main>
