<script>
const $ = id => document.getElementById(id);

function validateEmail(v){return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(v);}

function setError(fcId,show){
  const el=$(fcId);
  el.classList.toggle('has-error',show);
  el.classList.toggle('has-success',!show);
}

function togglePw(){
  const inp=$('sifre'),icon=$('toggle-pw');
  if(inp.type==='password'){inp.type='text';icon.className='bi bi-eye-slash field-icon-right';}
  else{inp.type='password';icon.className='bi bi-eye field-icon-right';}
}

function showToast(msg,icon='bi-check-circle-fill'){
  const wrap=$('toast-wrap');
  const t=document.createElement('div');
  t.className='toast-item';
  t.innerHTML=`<i class="bi ${icon}"></i><span>${msg}</span>`;
  wrap.appendChild(t);
  setTimeout(()=>{t.style.opacity='0';t.style.transform='translateX(40px)';t.style.transition='.3s';setTimeout(()=>t.remove(),300);},3500);
}

function showAlert(msg){
  const box=$('alert-box');
  $('alert-msg').textContent=msg;
  box.style.display='flex';
}
function hideAlert(){$('alert-box').style.display='none';}

$('eposta').addEventListener('blur',function(){
  if(this.value) setError('fc-eposta',!validateEmail(this.value.trim()));
});
$('sifre').addEventListener('blur',function(){
  if(this.value) setError('fc-sifre',this.value.length<6);
});
[$('eposta'),$('sifre')].forEach(el=>el.addEventListener('input',hideAlert));

$('login-form').addEventListener('submit',function(e){
  e.preventDefault();
  hideAlert();

  const eposta=$('eposta').value.trim();
  const sifre=$('sifre').value;
  let ok=true;

  const epostaOk=validateEmail(eposta);
  setError('fc-eposta',!epostaOk);
  if(!epostaOk) ok=false;

  const sifreOk=sifre.length>=6;
  setError('fc-sifre',!sifreOk);
  if(!sifreOk) ok=false;

  if(!ok){
    const firstErr=document.querySelector('.has-error input');
    if(firstErr) firstErr.scrollIntoView({behavior:'smooth',block:'center'});
    return;
  }

  const btn=$('submit-btn');
  btn.classList.add('loading');
  btn.disabled=true;

  // Simulate — replace with this.submit() in production
  setTimeout(()=>{
    btn.classList.remove('loading');
    btn.disabled=false;
    showToast('Giriş başarılı! Yönlendiriliyorsunuz...');
  },1800);
});

function socialClick(provider){
  showToast(`${provider} ile giriş deneniyor...`,'bi-person-check');
}
</script>
