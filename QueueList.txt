    <script>
        const QueueUI = (() => {

            // ── DOM Referansları ──
            const els = {
                card:        document.getElementById('q-current-card'),
                queueNum:    document.getElementById('q-queue-num-val'),
                patientName: document.getElementById('q-patient-name'),
                roomBadge:   document.getElementById('q-room-badge'),
                prevNum:     document.getElementById('q-prev-num'),
                prevName:    document.getElementById('q-prev-name'),
                statWaiting: document.getElementById('q-stat-waiting'),
                statCalled:  document.getElementById('q-stat-called'),
                statDone:    document.getElementById('q-stat-done'),
                barWaiting:  document.getElementById('q-bar-waiting'),
                barCalled:   document.getElementById('q-bar-called'),
                barDone:     document.getElementById('q-bar-done'),
                tbody:       document.getElementById('q-tbody'),
                tblCount:    document.getElementById('q-tbl-count'),
                connDot:     document.getElementById('q-conn-dot'),
                connText:    document.getElementById('q-conn-text'),
            };

            // ── Geçerli durum ──
            let state = {
                currentNum:  '042',
                currentName: 'Ayşe Fatma Kaya',
                currentRoom: '3',
                waiting:     14,
                called:      7,
                done:        21,
            };

            // ────────────────────────────────────────
            // updateCurrentPatient
            //   SignalR "PatientCalled" event handler'ından
            //   veya test fonksiyonundan çağrılır.
            // ────────────────────────────────────────
            function updateCurrentPatient(data) {
                /*
                 * Beklenen data yapısı:
                 * {
                 *   queueNumber : string  // örn. "043"
                 *   patientName : string  // örn. "Mustafa Çelik"
                 *   roomNumber  : string  // örn. "2"
                 *   waiting     : number
                 *   called      : number
                 *   done        : number
                 * }
                 */

                // Önceki hasta bilgisini sakla
                const prevNum  = state.currentNum;
                const prevName = state.currentName;

                // State güncelle
                state.currentNum  = data.queueNumber  ?? state.currentNum;
                state.currentName = data.patientName  ?? state.currentName;
                state.currentRoom = data.roomNumber   ?? state.currentRoom;
                state.waiting     = data.waiting      ?? state.waiting;
                state.called      = data.called       ?? state.called;
                state.done        = data.done         ?? state.done;

                // ── Sıra No güncellemesi ──
                els.queueNum.textContent = String(state.currentNum).padStart(3, '0');

                // ── Hasta Adı güncellemesi ──
                els.patientName.textContent = state.currentName;

                // ── Oda yönlendirmesi güncellemesi ──
                els.roomBadge.innerHTML =
                    `<i class="bi bi-door-open"></i>` +
                    ` Lütfen <strong style="margin:0 4px">${state.currentRoom} Numaralı Muayene Odası</strong>'na geçiniz`;

                // ── Önceki hasta bilgisi ──
                els.prevNum.textContent  = 'No ' + String(prevNum).padStart(3, '0');
                els.prevName.textContent = prevName;

                // ── İstatistik güncellemesi ──
                updateStats(state.waiting, state.called, state.done);

                // ── Kart animasyonu ──
                highlightCard();

                // ── Tabloda ilgili satırı güncelle ──
                updateTableRow(state.currentNum, state.currentName);
            }

            // ── İstatistik sayaç animasyonu ──
            function updateStats(waiting, called, done) {
                const total = waiting + called + done;

                animateCounter(els.statWaiting, waiting);
                animateCounter(els.statCalled,  called);
                animateCounter(els.statDone,    done);

                els.barWaiting.style.width = total ? Math.round((waiting / total) * 100) + '%' : '0%';
                els.barCalled.style.width  = total ? Math.round((called  / total) * 100) + '%' : '0%';
                els.barDone.style.width    = total ? Math.round((done    / total) * 100) + '%' : '0%';
            }

            function animateCounter(el, target) {
                if (!el) return;
                const start   = parseInt(el.textContent) || 0;
                const diff    = target - start;
                const steps   = 20;
                const delay   = 600 / steps;
                let current   = start;
                let step      = 0;

                el.classList.remove('q-count-animate');
                void el.offsetWidth; // reflow
                el.classList.add('q-count-animate');

                const timer = setInterval(() => {
                    step++;
                    current = Math.round(start + diff * (step / steps));
                    el.textContent = current;
                    if (step >= steps) { el.textContent = target; clearInterval(timer); }
                }, delay);
            }

            // ── Kart yanıp-sönme efekti ──
            function highlightCard() {
                els.card.classList.remove('q-flash');
                void els.card.offsetWidth; // reflow
                els.card.classList.add('q-flash');
                setTimeout(() => els.card.classList.remove('q-flash'), 2000);
            }

            // ── Tabloda satır durumunu güncelle ──
            function updateTableRow(queueNum, patientName) {
                if (!els.tbody) return;
                const formatted = 'No ' + String(queueNum).padStart(3, '0');

                // Önce tüm "called" satırları "done" yap
                els.tbody.querySelectorAll('.q-pill.called').forEach(pill => {
                    pill.className   = 'q-pill done';
                    pill.textContent = 'Tamamlandı';
                    const ticket = pill.closest('tr')?.querySelector('.q-ticket-num');
                    if (ticket) ticket.classList.remove('active');
                });

                // Şu anki sıra numarasını bul ve "called" yap
                els.tbody.querySelectorAll('.q-ticket-num').forEach(ticket => {
                    if (ticket.textContent.trim() === formatted) {
                        ticket.classList.add('active');
                        const pill = ticket.closest('tr')?.querySelector('.q-pill');
                        if (pill) {
                            pill.className   = 'q-pill called';
                            pill.textContent = 'Çağrıldı';
                        }
                    }
                });
            }

            // ── Bağlantı durumu güncelleme ──
            function setConnectionStatus(status) {
                // status: 'connecting' | 'connected' | 'disconnected'
                const dot  = els.connDot;
                const text = els.connText;
                if (!dot || !text) return;

                dot.className = 'q-conn-dot ' + status;
                if (status === 'connected')    text.textContent = 'Bağlı';
                if (status === 'disconnected') text.textContent = 'Bağlantı Kesildi';
                if (status === 'connecting')   text.textContent = 'Bağlanıyor...';
            }

            // ── Public API ──
            return { updateCurrentPatient, updateStats, setConnectionStatus, highlightCard };
        })();
    </script>

    <!-- ════════════════════════════════════════════
         SignalR BAĞLANTISI
         --------------------------------------------
         Bu bölümde SignalR Hub URL'si ve event
         listener'ları yapılandırılır.
         Production'da Hub URL'si appsettings.json
         veya Razor ViewBag üzerinden alınmalıdır.
    ════════════════════════════════════════════ -->
    <script>
        (function () {

            // ── SignalR Hub URL — buraya kendi Hub adresinizi yazın ──
            // Örnek: const HUB_URL = "/hubs/queue";
            const HUB_URL = "/hubs/queue"; // SignalR Hub URL will be added here

            // ── HubConnection oluştur ──
            const connection = new signalR.HubConnectionBuilder()
                .withUrl(HUB_URL)
                .withAutomaticReconnect({
                    // Bağlantı koptuğunda yeniden deneme aralıkları (ms)
                    nextRetryDelayInMilliseconds: retryContext => {
                        if (retryContext.elapsedMilliseconds < 10000) return 2000;
                        if (retryContext.elapsedMilliseconds < 60000) return 5000;
                        return 10000;
                    }
                })
                .configureLogging(signalR.LogLevel.Information)
                .build();

            // ────────────────────────────────────────
            // "PatientCalled" Event Listener
            //
            // Hub tarafında şu şekilde çağrılmalıdır:
            //   await Clients.All.SendAsync("PatientCalled", new {
            //       QueueNumber = "043",
            //       PatientName = "Mustafa Çelik",
            //       RoomNumber  = "2",
            //       Waiting     = 13,
            //       Called      = 8,
            //       Done        = 21
            //   });
            // ────────────────────────────────────────
            connection.on("PatientCalled", function (data) {
                // Queue Number update
                // Patient Name update
                // Statistics update
                // Card highlight animation
                QueueUI.updateCurrentPatient({
                    queueNumber: data.queueNumber,   // Queue Number update
                    patientName: data.patientName,   // Patient Name update
                    roomNumber:  data.roomNumber,
                    waiting:     data.waiting,       // Waiting count update
                    called:      data.called,        // Called count update
                    done:        data.done           // Done count update
                });
            });

            // ── StatisticsUpdated Event (opsiyonel — sadece istatistik güncellemesi) ──
            connection.on("StatisticsUpdated", function (data) {
                QueueUI.updateStats(data.waiting, data.called, data.done);
            });

            // ── Bağlantı olayları ──
            connection.onreconnecting(() => {
                QueueUI.setConnectionStatus('connecting');
                console.log("[Queue] SignalR yeniden bağlanıyor...");
            });

            connection.onreconnected(() => {
                QueueUI.setConnectionStatus('connected');
                console.log("[Queue] SignalR yeniden bağlandı.");
            });

            connection.onclose(() => {
                QueueUI.setConnectionStatus('disconnected');
                console.warn("[Queue] SignalR bağlantısı kapandı.");
            });

            // ── Bağlantıyı başlat ──
            async function startConnection() {
                try {
                    QueueUI.setConnectionStatus('connecting');
                    await connection.start();
                    QueueUI.setConnectionStatus('connected');
                    console.log("[Queue] SignalR bağlantısı kuruldu. Hub:", HUB_URL);
                } catch (err) {
                    QueueUI.setConnectionStatus('disconnected');
                    console.error("[Queue] SignalR bağlantı hatası:", err);
                    // 5 saniye sonra tekrar dene
                    setTimeout(startConnection, 5000);
                }
            }

            startConnection();

            // ── Global erişim (geliştirme/test için) ──
            window._queueConnection = connection;

        })();
    </script>
